135// Conversions between SL_Bus_DOB_ros_std_msgs_Header and std_msgs::Header



136



137void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_DOB_ros_std_msgs_Header const* busPtr)



138{



139  const std::string rosMessageType("std_msgs/Header");



140



141  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);



142  msgPtr->seq =  busPtr->Seq;



143  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



144}



145



146void convertToBus(SL_Bus_DOB_ros_std_msgs_Header* busPtr, std_msgs::Header const* msgPtr)



147{



148  const std::string rosMessageType("std_msgs/Header");



149



150  convertToBusVariablePrimitiveArray(busPtr->FrameId, busPtr->FrameId_SL_Info, msgPtr->frame_id, slros::EnabledWarning(rosMessageType, "frame_id"));



151  busPtr->Seq =  msgPtr->seq;



152  convertToBus(&busPtr->Stamp, &msgPtr->stamp);



153}
