134



135void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_DOB_ros_std_msgs_Header const* busPtr)



136{



137  const std::string rosMessageType("std_msgs/Header");



138



139  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);



140  msgPtr->seq =  busPtr->Seq;



141  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



142}

144void convertToBus(SL_Bus_DOB_ros_std_msgs_Header* busPtr, std_msgs::Header const* msgPtr)



145{



146  const std::string rosMessageType("std_msgs/Header");



147



148  convertToBusVariablePrimitiveArray(busPtr->FrameId, busPtr->FrameId_SL_Info, msgPtr->frame_id, slros::EnabledWarning(rosMessageType, "frame_id"));



149  busPtr->Seq =  msgPtr->seq;



150  convertToBus(&busPtr->Stamp, &msgPtr->stamp);
