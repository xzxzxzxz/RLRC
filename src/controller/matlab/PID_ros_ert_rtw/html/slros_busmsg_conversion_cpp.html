116// Conversions between SL_Bus_PID_ros_std_msgs_Header and std_msgs::Header



117



118void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_PID_ros_std_msgs_Header const* busPtr)



119{



120  const std::string rosMessageType("std_msgs/Header");



121



122  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);



123  msgPtr->seq =  busPtr->Seq;



124  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



125}



126



127void convertToBus(SL_Bus_PID_ros_std_msgs_Header* busPtr, std_msgs::Header const* msgPtr)



128{



129  const std::string rosMessageType("std_msgs/Header");



130



131  convertToBusVariablePrimitiveArray(busPtr->FrameId, busPtr->FrameId_SL_Info, msgPtr->frame_id, slros::EnabledWarning(rosMessageType, "frame_id"));



132  busPtr->Seq =  msgPtr->seq;



133  convertToBus(&busPtr->Stamp, &msgPtr->stamp);



134}
